    <script>
        // --- SETTINGS ---
        let isTimerEnabled = true;

        // --- GAME STATE ---
        let totalScore = 0; 
        let roundScore = 0; 
        let correctAnswers = 0;
        
        let currentBits = 4;
        let pointsPerQ = 0;
        let currentDifficulty = 'easy'; 
        let timeLimit = 10; 
        
        let questionList = []; 
        let currentQIndex = 0; 
        let currentTarget = 0; 

        // --- TIMING & METRICS ---
        let timerInterval;
        let timeLeft = 0;
        let questionStartTime = 0;
        let runTotalTimeTaken = 0; 
        let runTotalQuestions = 0;

        let isProcessing = false; 
        let isPaused = false; 

        // --- MENU & NAVIGATION ---

        function toggleTimerMode() {
            isTimerEnabled = !isTimerEnabled;
            const btn = document.getElementById('timerToggleBtn');
            if (isTimerEnabled) {
                btn.innerText = "Timer: ON";
                btn.classList.remove('toggle-off');
                btn.classList.add('toggle-on');
            } else {
                btn.innerText = "Timer: OFF";
                btn.classList.remove('toggle-on');
                btn.classList.add('toggle-off');
            }
        }

        // Start New Run
        function startGame(difficulty) {
            totalScore = 0; 
            runTotalTimeTaken = 0;
            runTotalQuestions = 0;
            document.getElementById('totalScore').innerText = "0";
            startRound(difficulty);
        }

        // Continue Run
        function nextDifficulty() {
            if (currentDifficulty === 'easy') startRound('medium');
            else if (currentDifficulty === 'medium') startRound('hard');
        }

        function playAgain() {
            startRound(currentDifficulty);
        }

        // --- UNIFIED EXIT & SAVE LOGIC ---
        function saveAndExit() {
            clearInterval(timerInterval);
            saveToLeaderboard();
            resetToMenu();
        }

        function resetToMenu() {
            document.getElementById('pauseOverlay').classList.add('hidden');
            document.getElementById('game').classList.add('hidden');
            document.getElementById('summary').classList.add('hidden');
            document.getElementById('leaderboardScreen').classList.add('hidden');
            document.getElementById('menu').classList.remove('hidden');
            document.body.classList.remove('critical-glow');
        }

        // --- LEADERBOARD LOGIC ---

        function saveToLeaderboard() {
            // Only save if the player has actually played a bit
            if (totalScore === 0 && runTotalQuestions === 0) return;

            let avgTime = runTotalQuestions > 0 ? (runTotalTimeTaken / runTotalQuestions).toFixed(2) : "0.00";
            
            let entry = {
                score: totalScore,
                avgTime: avgTime,
                date: new Date().toLocaleDateString()
            };

            let leaderboard = JSON.parse(localStorage.getItem('binaryLeaderboard')) || [];
            leaderboard.push(entry);
            
            // Sort by Score (High to Low)
            leaderboard.sort((a, b) => b.score - a.score);
            
            localStorage.setItem('binaryLeaderboard', JSON.stringify(leaderboard));
        }

        function showLeaderboard() {
            document.getElementById('menu').classList.add('hidden');
            document.getElementById('leaderboardScreen').classList.remove('hidden');
            
            let tbody = document.getElementById('leaderboardBody');
            tbody.innerHTML = "";

            let leaderboard = JSON.parse(localStorage.getItem('binaryLeaderboard')) || [];

            if (leaderboard.length === 0) {
                tbody.innerHTML = "<tr><td colspan='4' class='empty-msg'>No games played yet!</td></tr>";
                return;
            }

            leaderboard.forEach((entry, index) => {
                let row = `
                    <tr>
                        <td>${index + 1}</td>
                        <td>${entry.score}</td>
                        <td>${entry.avgTime}s</td>
                        <td>${entry.date}</td>
                    </tr>
                `;
                tbody.innerHTML += row;
            });
        }

        function closeLeaderboard() {
            document.getElementById('leaderboardScreen').classList.add('hidden');
            document.getElementById('menu').classList.remove('hidden');
        }

        // --- PAUSE LOGIC ---

        function pauseGame() {
            if(isProcessing) return; 
            isPaused = true;
            clearInterval(timerInterval); 
            document.getElementById('pauseOverlay').classList.remove('hidden'); 
        }

        function resumeGame() {
            isPaused = false;
            document.getElementById('pauseOverlay').classList.add('hidden'); 
            startTimer(); 
        }

        // --- ROUND SETUP ---

        function startRound(difficulty) {
            currentDifficulty = difficulty;
            
            if (difficulty === 'easy') {
                currentBits = 4; pointsPerQ = 400; timeLimit = 10;
            } else if (difficulty === 'medium') {
                currentBits = 6; pointsPerQ = 700; timeLimit = 15;
            } else {
                currentBits = 8; pointsPerQ = 1000; timeLimit = 30;
            }

            questionList = generateUniqueNumbers(currentBits, 8);
            
            currentQIndex = 0;
            roundScore = 0;
            correctAnswers = 0;
            
            document.getElementById('menu').classList.add('hidden');
            document.getElementById('summary').classList.add('hidden');
            document.getElementById('game').classList.remove('hidden');

            createCards(currentBits);
            loadQuestion();
        }

        function generateUniqueNumbers(bits, count) {
            let max = Math.pow(2, bits) - 1;
            let numbers = new Set();
            while(numbers.size < count) {
                let r = Math.floor(Math.random() * max) + 1;
                numbers.add(r);
            }
            return Array.from(numbers);
        }

        function createCards(bits) {
            const container = document.getElementById('cardContainer');
            container.innerHTML = ''; 

            for (let i = bits - 1; i >= 0; i--) {
                let value = Math.pow(2, i);
                let cardHTML = `
                    <div class="card" data-value="${value}" onclick="toggleCard(this)">
                        <div class="card-inner">
                            <div class="card-front">
                                <span class="digit">0</span>
                                <span class="label">${value}</span>
                            </div>
                            <div class="card-back">
                                <span class="digit">1</span>
                                <span class="label">${value}</span>
                            </div>
                        </div>
                    </div>
                `;
                container.innerHTML += cardHTML;
            }
        }

        // --- GAME LOOP ---

        function loadQuestion() {
            if (currentQIndex >= 8) {
                endRound();
                return;
            }

            isProcessing = false;
            isPaused = false;
            
            document.body.classList.remove('critical-glow');
            document.body.classList.remove('success-glow');

            currentTarget = questionList[currentQIndex];
            document.getElementById('targetNum').innerText = currentTarget;
            document.getElementById('qIndex').innerText = currentQIndex + 1;
            document.getElementById('message').innerText = "";
            document.querySelector('.submit-btn').disabled = false;
            
            let cards = document.querySelectorAll('.card');
            cards.forEach(card => card.classList.remove('active'));

            questionStartTime = Date.now();

            if (isTimerEnabled) {
                timeLeft = timeLimit;
                document.getElementById('timer').innerText = timeLeft;
                startTimer();
            } else {
                document.getElementById('timer').innerText = "‚àû";
                clearInterval(timerInterval); 
            }
        }

        function startTimer() {
            if (!isTimerEnabled) return;

            clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                if(!isPaused) {
                    timeLeft--;
                    document.getElementById('timer').innerText = timeLeft;
                    
                    if (timeLeft < 4 && timeLeft > 0) {
                        document.body.classList.add('critical-glow');
                    }

                    if (timeLeft <= 0) {
                        clearInterval(timerInterval);
                        checkAnswer(true); 
                    }
                }
            }, 1000);
        }

        function toggleCard(card) {
            if (!isProcessing && !isPaused) {
                card.classList.toggle('active');
            }
        }

        function checkAnswer(isAutoSubmit) {
            if (isProcessing || isPaused) return; 
            isProcessing = true;
            clearInterval(timerInterval);
            document.body.classList.remove('critical-glow'); 
            document.querySelector('.submit-btn').disabled = true;

            // Stats for Leaderboard
            let timeSpent = (Date.now() - questionStartTime) / 1000; 
            if (isTimerEnabled && timeSpent > timeLimit) timeSpent = timeLimit;
            
            runTotalTimeTaken += timeSpent;
            runTotalQuestions++;

            let total = 0;
            let activeCards = document.querySelectorAll('.card.active');
            activeCards.forEach(card => total += parseInt(card.getAttribute('data-value')));

            let msg = document.getElementById('message');
            let isCorrect = (total === currentTarget);

            if (isCorrect) {
                if (isAutoSubmit) {
                    msg.innerText = "Time's Up! Auto-submit Correct!";
                } else {
                    msg.innerText = "Correct!";
                }
                msg.style.color = "#43a047";
                document.body.classList.add('success-glow');

                totalScore += pointsPerQ;
                roundScore += pointsPerQ;
                correctAnswers++;
                document.getElementById('totalScore').innerText = totalScore;
            } else {
                let correctBinary = currentTarget.toString(2);
                correctBinary = correctBinary.padStart(currentBits, '0');

                if (isAutoSubmit) {
                    msg.innerText = `Time's Up! Wrong! ${currentTarget} is [${correctBinary}]`;
                } else {
                    msg.innerText = `Wrong! ${currentTarget} is [${correctBinary}]`;
                }
                msg.style.color = "#e91e63";
            }

            setTimeout(() => {
                currentQIndex++;
                loadQuestion();
            }, 2500);
        }

        function endRound() {
            document.getElementById('game').classList.add('hidden');
            document.getElementById('summary').classList.remove('hidden');
            document.body.classList.remove('critical-glow');
            document.body.classList.remove('success-glow');
            
            document.getElementById('correctCount').innerText = correctAnswers;
            document.getElementById('roundPoints').innerText = roundScore;

            const nextBtn = document.getElementById('btnNextLevel');
            if (currentDifficulty === 'hard') {
                nextBtn.style.display = 'none'; 
            } else {
                nextBtn.style.display = 'block';
                let nextName = (currentDifficulty === 'easy') ? "Medium" : "Hard";
                nextBtn.innerText = "Play " + nextName;
            }
        }
    </script>
